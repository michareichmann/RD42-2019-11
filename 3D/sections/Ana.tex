\subsection{Event Cuts}
\begin{frame}{Cuts}

	\nicetab{ll}{ \good{Cut} & \good{Excluded Events} 														\\\hline

	rhit								& large DUT residual (\SI{>100}{\milli\meter})						\\
	pixel mask					& noisy pixels																						\\
	fiducial						& not in selected (fiducial) area of the DUT							\\
